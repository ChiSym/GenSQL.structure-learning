----

[source,iql]
----
SELECT
    PROBABILITY OF Sodium
        UNDER baseline_model
        GIVEN
            Potassium AND
            Albumin AND
            Alkaline_Phosphatase AND
            Potassium AND
            Alanine_Amino_Transferase
                AS prob,
    Sodium
FROM data
LIMIT 1000
----
