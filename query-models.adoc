[source,iql]

[source,iql]

SELECT * FROM (GENERATE * UNDER baseline_model) LIMIT 10
----

[source,iql]
----
SELECT
    PROBABILITY OF Salary_USD
        UNDER baseline_model
        GIVEN
            Age
         AS prob,
    Salary_USD,
    Age
FROM data
WHERE Salary_USD < 500000
LIMIT 1000
----
